// cmds for camera capturing va network

sudo apt update
sudo apt install ffmpeg vlc

//in laptop
ffmpeg -f v4l2 -i /dev/video0 -pix_fmt yuv420p -r 30 -f rtsp -rtsp_transport tcp rtsp://<laptop_ip_address>:8554/stream

//in raspberry pi
cvlc rtsp://<laptop_ip_address>:8554/stream --sout '#rtp{sdp=rtsp://:8554/}' --demux h264
